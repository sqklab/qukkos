function(add_qukkos_qsharp_test test_name relative_qs_source_location relative_cpp_source_location)
  add_test(
  NAME
    ${test_name}
  COMMAND
    bash -c "${CMAKE_BINARY_DIR}/qukkos -qrt ftqc -qs-build-exe ${CMAKE_CURRENT_SOURCE_DIR}/${relative_qs_source_location} ${CMAKE_CURRENT_SOURCE_DIR}/${relative_cpp_source_location} -o ${test_name}; \
              ${CMAKE_CURRENT_BINARY_DIR}/${test_name}"
  )
endfunction()

add_qukkos_qsharp_test(qukkos_qsharp_bell FTQC/Bell/bell.qs FTQC/Bell/bell_driver.cpp)
add_qukkos_qsharp_test(qukkos_qsharp_deuteron FTQC/Deuteron/vqe_ansatz.qs FTQC/Deuteron/vqe_driver.cpp)
add_qukkos_qsharp_test(qukkos_qsharp_functor FTQC/Functor/functor.qs FTQC/Functor/driver.cpp)
add_qukkos_qsharp_test(qukkos_qsharp_qpe FTQC/qpe/qpe.qs FTQC/qpe/driver.cpp)
